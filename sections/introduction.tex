\subsubsection{Granulare Funktions-Markierung}

\textbf{Problem:} Wie markiert man einzelne Funktionen in gemischtem Code?

\textbf{Lösung 1: Kommentare im Code selbst}
\begin{verbatim}
# Manuell geschrieben
def load_data(filename):
    with open(filename, 'r') as f:
        return f.read()

def process_text(text):  # <- KI-generiert (ChatGPT): process_text
    words = text.split()
    return len(words)

# Wieder manuell geschrieben
def save_result(result, filename):
    with open(filename, 'w') as f:
        f.write(str(result))

result = sum(data)  # <- KI (Copilot)
\end{verbatim}

\textbf{Lösung 2: LaTeX-Befehle nach Code-Blöcken}

\begin{verbatim}
def process_text(text):
    words = text.split()
    return len(words)
\end{verbatim}
\aifunction{ChatGPT}{Erstelle Textverarbeitungsfunktion}{20.03.2024}{process\_text}

\begin{verbatim}
result = sum(data)
\end{verbatim}
\ailine{Copilot}{Berechne Summe}{21.03.2024}

\textbf{Lösung 3: Kompletter KI-Block (empfohlen)}
\begin{aicode}{GitHub Copilot}{Erstelle mehrere Hilfsfunktionen}{22.03.2024}
\begin{verbatim}
def helper_function_1(data):
    return data * 2

def helper_function_2(result):
    return result + 1
\end{verbatim}
Diese Funktionen wurden komplett von KI generiert.
\end{aicode}

\textbf{Vorteile der neuen Lösung:}
\begin{itemize}
    \item \textbf{Visuelle Box:} KI-Code ist sofort erkennbar
    \item \textbf{Granulare Markierung:} Einzelne Funktionen markierbar
    \item \textbf{Inline-Kommentare:} Einzelne Zeilen markierbar
    \item \textbf{Flexibel:} Für alle Code-Szenarien geeignet
\end{itemize}

\subsection{Tipps und Tricks}

\begin{itemize}
    \item Verwenden Sie \verb|\cref{}| statt \verb|\ref{}| für bessere Verweise
    \item Alle Bilder sollten ein aussagekräftiges Label haben
    \item \textbf{KI-ID-System:} Verwenden Sie \verb|\cref{ai:1}| für KI-Referenzen
    \item \textbf{Sauberer Lesefluss:} Alle KI-Details nur in der hinteren Anlage
    \item \textbf{Keine Fußnoten-Überflutung:} KI-Befehle stören nicht mehr den Text
    \item \textbf{Prüferfreundlich:} Hyperlinks springen zur Verwendungsstelle
    \item KI-Kennzeichnung ist bei \verb|\enableAI| verpflichtend
    \item TODOs werden automatisch gesammelt und können mit \verb|\printtodos| ausgegeben werden
    \item Die Vorlage folgt exakt den BA Bautzen Richtlinien von 2025
    \item Verwenden Sie \verb|\medskip|, \verb|\bigskip| für manuelle Abstände
    \item Bei Problemen: Logs prüfen und schrittweise kompilieren
\end{itemize}
